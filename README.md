Setup steps:

1 - git clone https://github.com/odaydid002/Exam-Managment-Project.git

2 - cd Exam-Managment-Project

3 - cd backend

4 - composer install

5 - cp .env.example .env

6 - php artisan key:generate

7 - php artisan migrate

8 - cd ..

9 - cd frontend

10 - npm i

11 - npm run build
