Setup steps:

  1 - git clone https://github.com/username/Exam-Managment-Project.git \n
  2 - cd Exam-Managment-Project \n
  3 - cd backend \n
  4 - composer install \n
  5 - cp .env.example .env \n
  6 - php artisan key:generate \n
  7 - php artisan migrate \n
  8 - cd .. \n
  9 - cd frontend \n
  10 - npm i \n
  11 - npm run build \n
